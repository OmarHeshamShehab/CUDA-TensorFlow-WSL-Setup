#!/bin/bash

set -e

# âŒ Prevent running as root
if [ "$EUID" -eq 0 ]; then
  echo "âŒ ERROR: Do NOT run this script as root or with sudo."
  exit 1
fi

echo "âœ… Running as user: $USER"

# ğŸ”„ Update system
sudo apt update && sudo apt upgrade -y

# ğŸ§¹ Cleanup handler
CUDA_DEB="cuda-repo-wsl-ubuntu-12-5-local_12.5.0-1_amd64.deb"

trap 'rm -f $CUDA_DEB $CUDNN_DEB miniconda.sh 2>/dev/null || true' EXIT

# âš™ï¸ Install CUDA if not already present
if [ -d "/usr/local/cuda" ]; then
  echo "âš™ï¸ CUDA already installed â€” skipping."

  sudo mv cuda-wsl-ubuntu.pin /etc/apt/preferences.d/cuda-repository-pin-600
  wget --show-progress https://developer.download.nvidia.com/compute/cuda/12.5.0/local_installers/$CUDA_DEB
  sudo dpkg -i $CUDA_DEB
  sudo cp /var/cuda-repo-wsl-ubuntu-12-5-local/cuda-*-keyring.gpg /usr/share/keyrings/
  sudo apt update
  sudo apt install -y cuda
  echo "âœ… CUDA installed."
fi

# â• Install cuDNN 9.3.0 if not already installed
if dpkg -l | grep -q libcudnn9; then
  echo "âœ… cuDNN 9.3 already installed â€” skipping."
else
  echo "â• Installing cuDNN 9.3.0 (Ubuntu 24.04)..."
  wget --show-progress https://developer.download.nvidia.com/compute/cudnn/9.3.0/local_installers/$CUDNN_DEB
  sudo dpkg -i $CUDNN_DEB
  sudo cp /var/cudnn-local-repo-ubuntu2404-9.3.0/cudnn-*-keyring.gpg /usr/share/keyrings/
  sudo apt-get update
  sudo apt-get install -y cudnn libcudnn-dev
  echo "âœ… cuDNN and dev headers installed."
fi

# â• Add CUDA paths to ~/.bashrc if missing
if ! grep -q "/usr/local/cuda/bin" ~/.bashrc; then
  echo "â• Adding CUDA environment variables to ~/.bashrc..."
  echo 'export PATH=/usr/local/cuda/bin:$PATH' >> ~/.bashrc
  echo 'export LD_LIBRARY_PATH=/usr/local/cuda/lib64:$LD_LIBRARY_PATH' >> ~/.bashrc
else
  echo "âœ… CUDA paths already exist in ~/.bashrc â€” skipping."
fi

# Apply CUDA paths immediately
export PATH=/usr/local/cuda/bin:$PATH
export LD_LIBRARY_PATH=/usr/local/cuda/lib64:$LD_LIBRARY_PATH

# ğŸ“¥ Install Miniconda if not already installed
if [ -f "$HOME/miniconda/bin/conda" ]; then
  echo "âœ… Miniconda already installed â€” skipping download."
else
  echo "ğŸ“¥ Installing Miniconda..."
  wget --show-progress -O miniconda.sh https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh
  bash miniconda.sh -b -p "$HOME/miniconda"
fi

# Ensure Conda path is in bashrc and current session
echo 'export PATH="$HOME/miniconda/bin:$PATH"' >> ~/.bashrc
export PATH="$HOME/miniconda/bin:$PATH"

# ğŸ” Initialize Conda & disable auto base
$HOME/miniconda/bin/conda init bash
source ~/.bashrc
conda config --set auto_activate_base false

# ğŸ”„ Update Conda base installation
echo "ğŸ”„ Updating Conda base installation..."

# ğŸ Create Conda env
echo "ğŸ Creating Conda environment '.tf219' with Python 3.11..."

# âœ… Activate safely
echo "âœ… Activating .tf219 environment..."
source "$HOME/miniconda/etc/profile.d/conda.sh"

# ğŸ“¦ Install C++ stdlib fix for pycuda
echo "ğŸ“¦ Installing libstdcxx-ng (for GLIBCXX_3.4.32 support)..."
conda install -c conda-forge libstdcxx-ng -y

# ğŸ“¦ Install TensorFlow with GPU support
echo "ğŸ“¦ Installing TensorFlow with GPU support..."
pip install "tensorflow[and-cuda]"

# ğŸ“¦ Install pycuda and TensorRT
echo "ğŸ“¦ Installing pycuda and tensorrt..."
pip install pycuda
pip install nvidia-pyindex
pip install tensorrt

# ğŸ“¦ Install GPU CLI tools (optional)
echo "ğŸ“¦ Installing GPU CLI tools (nvidia-smi)..."
sudo apt install -y nvidia-utils-525 || true

# ğŸ§¹ Clean APT cache
echo "ğŸ§¹ Cleaning APT package cache..."

# âœ… Done
echo ""
echo "ğŸ‰ Full setup complete!"
