#!/bin/bash
# ğŸš€ TensorFlow 2.20 GPU Installer (Universal for Ubuntu 20.04, 22.04, 24.04)

set -e

# âŒ Prevent running as root

# âš¡ Detect Ubuntu version

# âš¡ Detect GPU (basic)

  wget --show-progress https://developer.download.nvidia.com/compute/cuda/12.5.0/local_installers/$CUDA_DEB
  sudo dpkg -i $CUDA_DEB
  sudo cp /var/cuda-repo-wsl-ubuntu-12-5-local/cuda-*-keyring.gpg /usr/share/keyrings/

  sudo apt install -y cuda
  echo "âœ… CUDA 12.5 installed."

# â• Install cuDNN 9.3 (per Ubuntu version)

  echo "âœ… cuDNN 9.3 already installed â€” skipping."

  case "$UBUNTU_VERSION" in

      ;;
    22.04)
      CUDNN_DEB="cudnn-local-repo-ubuntu2204-9.3.0_1.0-1_amd64.deb"
      ;;
    24.04)
      CUDNN_DEB="cudnn-local-repo-ubuntu2404-9.3.0_1.0-1_amd64.deb"
      ;;
    *)
      echo "âŒ Unsupported Ubuntu version: $UBUNTU_VERSION"

  wget --show-progress https://developer.download.nvidia.com/compute/cudnn/9.3.0/local_installers/$CUDNN_DEB
  sudo dpkg -i $CUDNN_DEB
  sudo cp /var/cudnn-local-repo-ubuntu*/cudnn-*-keyring.gpg /usr/share/keyrings/
  sudo apt update
  sudo apt install -y cudnn libcudnn-dev
  echo "âœ… cuDNN 9.3 installed."

# â• Add CUDA paths to ~/.bashrc if missing

# Apply CUDA paths immediately

# ğŸ“¥ Install Miniconda if not already installed

# Ensure Conda path
echo 'export PATH="$HOME/miniconda/bin:$PATH"' >> ~/.bashrc
export PATH="$HOME/miniconda/bin:$PATH"

# ğŸ” Initialize Conda
$HOME/miniconda/bin/conda init bash
source ~/.bashrc
conda config --set auto_activate_base false

# â• Accept Conda Terms of Service
conda tos accept --override-channels --channel https://repo.anaconda.com/pkgs/main
conda tos accept --override-channels --channel https://repo.anaconda.com/pkgs/r

# ğŸ”„ Update Conda
conda update -n base -c defaults conda -y

# ğŸ Create Conda env

# âœ… Activate safely

# ğŸ“¦ Install C++ stdlib fix

# ğŸ“¦ Install TensorFlow 2.20 GPU

# ğŸ“¦ Install pycuda + TensorRT

# ğŸ“¦ Install common packages

# ğŸ“¦ Ensure nvidia-smi

  echo "âœ… nvidia-smi already available â€” skipping."

    | awk '{print $1}' \
    | sort -V \

  else
    echo "âš ï¸ Could not auto-detect nvidia-utils package â€” skipping."
  fi
fi

# ğŸ§¹ Clean APT cache

# âœ… Done
echo ""
echo "ğŸ‰ Full TensorFlow 2.20 GPU setup complete!"
echo "ğŸ‘‰ To activate later: conda activate .tf220"
echo "ğŸ§ª To test GPU in Python:"
